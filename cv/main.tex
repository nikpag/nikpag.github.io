	{
		\tl_if_blank:VF \l_cv_section_line_tl
			{
				\seq_put_right:NV \l_cv_section_lines_seq \l_cv_section_line_tl
				\tl_clear:N \l_cv_section_line_tl
			}
	}
\cs_new_protected:Npn \setCvSectionHeader #1
	{
		\seq_set_split:Nnn \l_cv_section_words_seq {~} {#1}
		\seq_clear:N \l_cv_section_lines_seq
		\tl_clear:N \l_cv_section_line_tl
		\seq_map_inline:Nn \l_cv_section_words_seq
			{
				\tl_if_blank:VTF \l_cv_section_line_tl
					{ \tl_set:Nn \l_cv_section_candidate_tl {##1} }
					{ \tl_set:Nx \l_cv_section_candidate_tl { \l_cv_section_line_tl \c_space_tl ##1 } }
				\hbox_set:Nn \l_cv_section_measure_box { \sectionfont \l_cv_section_candidate_tl }
				\dim_compare:nNnTF { \box_wd:N \l_cv_section_measure_box } > { \cvsectionheaderwidth }
					{
						\tl_if_blank:VTF \l_cv_section_line_tl
							{ \tl_set:Nn \l_cv_section_line_tl {##1} }
							{
								\cv_section_store_current_line:
								\tl_set:Nn \l_cv_section_line_tl {##1}
							}
					}
					{ \tl_set_eq:NN \l_cv_section_line_tl \l_cv_section_candidate_tl }
			}
		\cv_section_store_current_line:
		\setbox\cvsectionheaderbox=\vtop
			{
				\parindent=0pt
				\seq_map_inline:Nn \l_cv_section_lines_seq
					{ \hbox { \sectionfont ##1 } }
			}
	}

% Apply margin settings after resume class initialization
\AtBeginDocument{%
  % Set LaTeX page geometry parameters
  % LaTeX adds 1in by default, so we subtract it to get the desired left margin
  \setlength{\oddsidemargin}{\cvleftmargin}%
  \addtolength{\oddsidemargin}{-1in}%
  \setlength{\evensidemargin}{\oddsidemargin}%
  
  % Calculate maximum available width for resume content
  % Available width = paper width - left margin - right margin
  \newlength{\availablewidth}%
  \setlength{\availablewidth}{8.5in}%
  \addtolength{\availablewidth}{-\cvleftmargin}%
  \addtolength{\availablewidth}{-\cvrightmargin}%
  
  % The resume class structure: resumewidth = sectionwidth + textwidth
  % We want the total resume width to fit within availablewidth
  % So: resumewidth = availablewidth (or less)
  \setlength{\resumewidth}{\availablewidth}%
  
  % Section layout is handled locally by the rSection environment.
  \setlength{\textwidth}{\resumewidth}%
  \setlength{\hsize}{\textwidth}%
  \setlength{\linewidth}{\textwidth}%
  \setlength{\hoffset}{0pt}%
}
